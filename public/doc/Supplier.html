def self.admin_grid(params = {})
  grid = Supplier
  #grid.includes(:product)
  grid = grid.where("suppliers.name LIKE ?", "#{params[:name]}%")   if params[:name].present?
  grid = grid.where("suppliers.email LIKE ?", "#{params[:email]}%") if params[:email].present?
  grid
end
